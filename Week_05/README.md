```
def debfs(start, end):
  if start == end:  # 必须判断
    return 2  # 根据题目 1 or 2
  s1, s2, visited = {start}, {end}, {start, end}
  level = 2  # 根据题目 1 or 2
  while s1:
    tmp = set()
    for cur in s1:
      for nex in gen_relative_nodes(cur):
        if nex in s2: return level  # found
        if nex not in visited:
          tmp.add(nex)
          visited.add(nex)
    s1 = tmp
    if len(s1) > len(s2): s1, s2 = s2, s1
  return 0  # 根据题目 0 or -1
  ```
